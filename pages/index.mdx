export async function getStaticProps() {
  const headers = {
    Authorization: `bearer ${process.env.GITHUB_TOKEN}`,
  };
  const username = "blankart";
  const body = {
    query: `query {
            user(login: "${process.env.GITHUB_USERNAME}") {
              name
              contributionsCollection {
                contributionCalendar {
                  colors
                  totalContributions
                  weeks {
                    contributionDays {
                      color
                      contributionCount
                      date
                      weekday
                    }
                    firstDay
                  }
                }
              }
            }
          }`,
  };
  const response = await fetch("https://api.github.com/graphql", {
    method: "POST",
    body: JSON.stringify(body),
    headers: headers,
  });
  const data = await response.json();
  const values = {};
  let totalContributions = 0;
  data.data.user.contributionsCollection.contributionCalendar.weeks.forEach(
    (week) =>
      week.contributionDays.forEach((day) => {
        values[day.date] = day.contributionCount;
        totalContributions += day.contributionCount;
      })
  );
  return {
    props: {
      values,
      until: Object.keys(values)[Object.keys(values).length - 1],
      totalContributions,
    },
    revalidate: 60 * 1000 * 1000 * 24,
  };
}
